# Theming

Cartino UI features a powerful theming system that allows you to customize every aspect of your components.

## Theme Configuration

Create a `cartino-ui.config.ts` file in your project root:

```ts [cartino-ui.config.ts]
export default {
  theme: {
    colors: {
      primary: 'blue',
      secondary: 'purple',
      success: 'green',
      warning: 'yellow',
      error: 'red',
    },
    components: {
      button: {
        base: 'font-medium transition-colors',
        variants: {
          solid: 'bg-{color}-500 text-white hover:bg-{color}-600',
          outline: 'border-2 border-{color}-500 text-{color}-500',
          ghost: 'text-{color}-500 hover:bg-{color}-50',
        },
        sizes: {
          sm: 'text-sm px-3 py-1.5',
          md: 'text-base px-4 py-2',
          lg: 'text-lg px-6 py-3',
        }
      }
    }
  }
}
```

## CSS Variables

Customize using CSS variables:

```css [assets/css/theme.css]
:root {
  /* Brand Colors */
  --cartino-primary: 220 90% 56%;
  --cartino-secondary: 280 65% 60%;
  
  /* Semantic Colors */
  --cartino-success: 142 71% 45%;
  --cartino-warning: 38 92% 50%;
  --cartino-error: 0 84% 60%;
  
  /* Typography */
  --cartino-font-sans: 'Inter', system-ui, sans-serif;
  --cartino-font-mono: 'Fira Code', 'Courier New', monospace;
  
  /* Spacing & Layout */
  --cartino-radius: 0.5rem;
  --cartino-shadow-sm: 0 1px 2px rgba(0, 0, 0, 0.05);
  --cartino-shadow-md: 0 4px 6px rgba(0, 0, 0, 0.1);
  --cartino-shadow-lg: 0 10px 15px rgba(0, 0, 0, 0.1);
}

.dark {
  --cartino-primary: 217 91% 60%;
  --cartino-secondary: 280 65% 65%;
}
```

## Component Overrides

Override individual component styles:

```vue
<template>
  <!-- Global config -->
  <CButton color="primary">
    Default Button
  </CButton>
  
  <!-- Per-instance override -->
  <CButton
    :ui="{
      base: 'custom-base-class',
      color: 'bg-gradient-to-r from-purple-500 to-pink-500',
      padding: 'px-8 py-4'
    }"
  >
    Custom Button
  </CButton>
</template>
```

## Design Tokens

Use semantic tokens for consistent theming:

```ts
export default {
  theme: {
    extend: {
      colors: {
        // Commerce-specific colors
        'product-new': '#10b981',
        'product-sale': '#ef4444',
        'product-out': '#6b7280',
        
        // Status colors
        'order-pending': '#f59e0b',
        'order-processing': '#3b82f6',
        'order-completed': '#10b981',
        'order-cancelled': '#ef4444',
      },
      spacing: {
        'product-gap': '1.5rem',
        'cart-padding': '1rem',
      }
    }
  }
}
```

## Dark Mode

Cartino UI supports automatic dark mode:

```vue
<template>
  <CThemeToggle />
  
  <!-- Components adapt automatically -->
  <CProductCard
    :product="product"
    class="dark:bg-gray-800"
  />
</template>
```

### Custom Dark Mode Logic

```ts
// composables/useTheme.ts
export const useCartinoTheme = () => {
  const colorMode = useColorMode()
  
  const isDark = computed(() => colorMode.value === 'dark')
  
  const toggle = () => {
    colorMode.preference = isDark.value ? 'light' : 'dark'
  }
  
  return {
    isDark,
    toggle,
    colorMode
  }
}
```

## Presets

Use built-in presets for quick setup:

```ts [cartino-ui.config.ts]
import { cartinoPresets } from '@cartino/ui'

export default {
  preset: cartinoPresets.minimal, // or 'shopify', 'modern', 'classic'
  
  theme: {
    // Override specific values
    colors: {
      primary: 'indigo'
    }
  }
}
```
