# Frontend Extensions

Customize and extend Cartino's admin interface with Vue 3, Inertia.js, and Cartino UI components.

## Admin Panel Architecture

Cartino's admin uses:
- **Vue 3** with Composition API
- **Inertia.js** for server-side routing
- **Cartino UI** component library
- **Tailwind CSS** for styling
- **Vite** for fast builds

## Custom Pages

### Creating an Admin Page

Create a Vue page component:

```vue
<!-- resources/js/Pages/Subscriptions/Index.vue -->
<template>
  <AdminLayout title="Subscriptions">
    <template #header>
      <ui-PageHeader>
        <ui-PageTitle>Subscriptions</CPageTitle>
        <template #actions>
          <ui-button
            :to="route('subscriptions.create')"
            icon="i-lucide-plus"
          >
            New Subscription
          </ui-button>
        </template>
      </CPageHeader>
    </template>

    <ui-card>
      <ui-table
        :columns="columns"
        :data="subscriptions.data"
        :loading="loading"
      >
        <template #cell-status="{ row }">
          <ui-Badge :color="statusColor(row.status)">
            {{ row.status }}
          </CBadge>
        </template>

        <template #cell-actions="{ row }">
          <ui-Dropdown>
            <ui-DropdownItem
              :to="route('subscriptions.show', row.id)"
              icon="i-lucide-eye"
            >
              View
            </CDropdownItem>
            <ui-DropdownItem
              icon="i-lucide-x-circle"
              @click="cancelSubscription(row)"
            >
              Cancel
            </CDropdownItem>
          </CDropdown>
        </template>
      </CTable>

      <template #footer>
        <ui-Pagination
          :current-page="subscriptions.current_page"
          :total="subscriptions.total"
          :per-page="subscriptions.per_page"
        />
      </template>
    </ui-card>
  </AdminLayout>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import { router } from '@inertiajs/vue3'
import AdminLayout from '@/Layouts/AdminLayout.vue'

interface Props {
  subscriptions: {
    data: any[]
    current_page: number
    total: number
    per_page: number
  }
}

const props = defineProps<Props>()

const columns = [
  { key: 'id', label: 'ID', sortable: true },
  { key: 'customer.name', label: 'Customer' },
  { key: 'plan.name', label: 'Plan' },
  { key: 'status', label: 'Status' },
  { key: 'ends_at', label: 'Ends At' },
  { key: 'actions', label: '' },
]

const loading = ref(false)

const statusColor = (status: string) => {
  const colors: Record<string, string> = {
    active: 'green',
    cancelled: 'red',
    expired: 'gray',
    trial: 'blue',
  }
  return colors[status] || 'gray'
}

const cancelSubscription = (subscription: any) => {
  if (confirm('Are you sure you want to cancel this subscription?')) {
    router.post(route('subscriptions.cancel', subscription.id))
  }
}
</script>
```

### Register Route

```php
// routes/web.php
use Inertia\Inertia;

Route::get('#', function () {
    return Inertia::render('Subscriptions/Index', [
        'subscriptions' => Subscription::paginate(20),
    ]);
})->name('subscriptions.index');
```

## Custom Components

### Creating a Component

```vue
<!-- resources/js/Components/SubscriptionCard.vue -->
<template>
  <ui-card>
    <ui-card-header>
      <div class="flex items-center justify-between">
        <div>
          <ui-card-title>{{ subscription.plan.name }}</ui-card-title>
          <ui-CardDescription>
            {{ subscription.customer.name }}
          </ui-cardDescription>
        </div>
        <ui-Badge :color="statusColor">
          {{ subscription.status }}
        </CBadge>
      </div>
    </ui-card-header>

    <ui-card-content>
      <dl class="grid grid-cols-2 gap-4">
        <div>
          <dt class="text-sm text-gray-500">Started</dt>
          <dd class="font-medium">
            {{ formatDate(subscription.started_at) }}
          </dd>
        </div>
        <div>
          <dt class="text-sm text-gray-500">Ends</dt>
          <dd class="font-medium">
            {{ formatDate(subscription.ends_at) }}
          </dd>
        </div>
      </dl>
    </ui-card-content>

    <ui-CardFooter>
      <ui-button
        v-if="subscription.status === 'active'"
        variant="outline"
        color="red"
        @click="$emit('cancel')"
      >
        Cancel Subscription
      </ui-button>
      <ui-button
        v-else-if="subscription.status === 'expired'"
        @click="$emit('renew')"
      >
        Renew Subscription
      </ui-button>
    </ui-cardFooter>
  </ui-card>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import { format } from 'date-fns'

interface Props {
  subscription: {
    id: number
    status: string
    plan: { name: string }
    customer: { name: string }
    started_at: string
    ends_at: string
  }
}

const props = defineProps<Props>()

const emit = defineEmits<{
  cancel: []
  renew: []
}>()

const statusColor = computed(() => {
  const colors: Record<string, string> = {
    active: 'green',
    cancelled: 'red',
    expired: 'gray',
  }
  return colors[props.subscription.status] || 'gray'
})

const formatDate = (date: string) => {
  return format(new Date(date), 'MMM dd, yyyy')
}
</script>
```

## Composables

Create reusable logic with composables:

```ts
// resources/js/composables/useSubscriptions.ts
import { ref, computed } from 'vue'
import { router } from '@inertiajs/vue3'

export const useSubscriptions = () => {
  const subscriptions = ref<any[]>([])
  const loading = ref(false)

  const activeSubscriptions = computed(() =>
    subscriptions.value.filter(s => s.status === 'active')
  )

  const loadSubscriptions = async () => {
    loading.value = true
    try {
      const response = await fetch('/api/subscriptions')
      subscriptions.value = await response.json()
    } finally {
      loading.value = false
    }
  }

  const cancelSubscription = (id: number) => {
    router.post(`/api/subscriptions/${id}/cancel`, {}, {
      onSuccess: () => {
        loadSubscriptions()
      }
    })
  }

  return {
    subscriptions,
    loading,
    activeSubscriptions,
    loadSubscriptions,
    cancelSubscription,
  }
}
```

Usage:

```vue
<script setup lang="ts">
import { onMounted } from 'vue'
import { useSubscriptions } from '@/composables/useSubscriptions'

const {
  subscriptions,
  loading,
  loadSubscriptions,
  cancelSubscription
} = useSubscriptions()

onMounted(loadSubscriptions)
</script>
```

## Forms

Build forms with validation:

```vue
<!-- resources/js/Pages/Subscriptions/Create.vue -->
<template>
  <AdminLayout title="Create Subscription">
    <ui-card>
      <ui-card-header>
        <ui-card-title>New Subscription</ui-card-title>
      </ui-card-header>

      <form @submit.prevent="submit">
        <ui-card-content class="space-y-4">
          <ui-FormGroup
            label="Customer"
            :error="form.errors.customer_id"
          >
            <ui-Select
              v-model="form.customer_id"
              :options="customers"
              placeholder="Select customer"
              searchable
            />
          </CFormGroup>

          <ui-FormGroup
            label="Plan"
            :error="form.errors.plan_id"
          >
            <ui-Select
              v-model="form.plan_id"
              :options="plans"
              placeholder="Select plan"
            />
          </CFormGroup>

          <ui-FormGroup
            label="Trial Days"
            :error="form.errors.trial_days"
          >
            <ui-Input
              v-model.number="form.trial_days"
              type="number"
              placeholder="0"
            />
          </CFormGroup>

          <ui-FormGroup>
            <ui-Checkbox
              v-model="form.send_welcome_email"
              label="Send welcome email"
            />
          </CFormGroup>
        </ui-card-content>

        <ui-CardFooter class="flex justify-end gap-2">
          <ui-button
            variant="outline"
            :href="route('subscriptions.index')"
          >
            Cancel
          </ui-button>
          <ui-button
            type="submit"
            :loading="form.processing"
          >
            Create Subscription
          </ui-button>
        </ui-cardFooter>
      </form>
    </ui-card>
  </AdminLayout>
</template>

<script setup lang="ts">
import { useForm } from '@inertiajs/vue3'
import AdminLayout from '@/Layouts/AdminLayout.vue'

interface Props {
  customers: Array<{ value: number; label: string }>
  plans: Array<{ value: number; label: string }>
}

const props = defineProps<Props>()

const form = useForm({
  customer_id: null,
  plan_id: null,
  trial_days: 0,
  send_welcome_email: true,
})

const submit = () => {
  form.post(route('subscriptions.store'))
}
</script>
```

## Plugins

Create Vue plugins:

```ts
// resources/js/plugins/currency.ts
import { App } from 'vue'

export default {
  install(app: App) {

    ) => {

      }).format(amount)
```

## State Management

Use Pinia for complex state:

```ts
// resources/js/stores/subscriptions.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useSubscriptionStore = defineStore('subscriptions', () => {
  const subscriptions = ref<Subscription[]>([])
  const loading = ref(false)

  const active = computed(() =>
    subscriptions.value.filter(s => s.status === 'active')
  )

  const fetchSubscriptions = async () => {
    loading.value = true
    try {
      const response = await fetch('/api/subscriptions')
      subscriptions.value = await response.json()
    } finally {
      loading.value = false
    }
  }

  return {
    subscriptions,
    loading,
    active,
    fetchSubscriptions,
  }
})
```
